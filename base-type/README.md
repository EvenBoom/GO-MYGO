## int
### 分类
1.int（有符号,长度取决于CPU,至少32位）</br>
2.int8,int16,int32,int64（有符号,固定长度）</br>
3.uint（无符号,长度取决于CPU,至少32位）</br>
4.uint8,uint16,uint32.uint64（无符号，固定长度）</br>
### 声明
int
-----------------------------------
1.var n int</br>
2.var n int8</br>
3.var n int16</br>
4.var n int32</br>
5.var n int64</br>

unt
-----------------------------------
1.var n uint</br>
2.var n uint8</br>
3.var n uint16</br>
4.var n uint32</br>
5.var n uint64</br>
### 初始化
int
-----------------------------------
1.var n int = 0 or n:=0</br>
2.var n int8 = 0</br>
3.var n int16 = 0</br>
4.var n int32 = 0</br>
5.var n int64 = 0</br>

uint
-----------------------------------
1.var n uint = 0</br>
2.var n uint8 = 0</br>
3.var n uint16 = 0</br>
4.var n uint32 = 0</br>
5.var n uint64 = 0</br>
## float
### 分类
1.float32</br>
2.float64</br>
### 声明
1.var f float32</br>
2.var f float64</br>
### 初始化
1.var f float64 = 0 or f:=0.0</br>
2.var f float32 =0</br>
